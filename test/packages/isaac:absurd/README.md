
## Absurd
Meteor package for https://github.com/krasimir/absurd.git

## Use
    
    var api = Absurd();
    api.add({
        body: {
            marginTop: '20px',
            p: {
                color: '#000'
            }
        }
    });

    api.compile(function (err, css) {
        // insert style
    });
